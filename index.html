function calculateAndShow() {

  if (!state.skin && state.skin !== 0) {

    alert("Please select a skin condition to continue.");

    return;

  }


  const typeMap = {Bouldering:10,Routes:8,Strength:10,Conditioning:5,Competition:15,Mixed:9};

  const typeBase = typeMap[state.sessionType]||10;

  const durMult = state.duration<60?.7:state.duration<90?1:state.duration<120?1.3:state.duration<150?1.6:state.duration<180?1.9:2.3;

  const intMod = 1+((state.intensity-5)*0.08);

  const limitMod = state.limitAttempts<=5?1:state.limitAttempts<=10?1.15:state.limitAttempts<=20?1.3:1.5;

  const fingerMod = {Jugs:0.9,Mixed:1.0,"Small crimps":1.1,"Max crimps":1.2}[state.fingerType]||1.0;

  const ageAdj = state.age>=45?1.6:state.age>=30?1.5:1.0;

  const trainingLoad = typeBase*durMult*intMod*limitMod*fingerMod*ageAdj;


  const credits = (state.sleepBefore*3)+(state.postMeal?2:0)+(state.protein?3:0)+Math.min(state.water,4)+(state.nap?5:0)+(state.walk?2:0)+(state.coldShower?2:0)+(state.stretch?1:0);

  const balance = credits - trainingLoad;


  const fingerScore = Math.max(state.fingerStiff,state.fingerFist,state.fingerCrimp,state.fingerHang);

  const forearmScore = (state.forearmSore+state.forearmTight+state.forearmPump)/3;

  const cnsScore = 10-(((state.motivation+state.explosive+state.energy+state.focus)/4)*2);

  const genScore = (state.sleepQuality+(10-state.stress))/2;


  const fingerR  = Math.max(0,Math.min(100,100-fingerScore*10));

  const forearmR = Math.max(0,Math.min(100,100-forearmScore*10));

  const genR     = Math.max(0,Math.min(100,genScore*10));

  const balBonus = Math.max(-20,Math.min(10,balance*0.5));

  ];

  const minPct = Math.min(...systems.map(s=>s.pct));

  document.getElementById("scoreBig").textContent = composite;

  document.getElementById("scoreBig").style.color = col;

  badge.style.color = col; badge.style.borderColor = col;

  badge.style.background = col+"22";


  const recBox = document.getElementById("recBox");

  recBox.style.background = col+"11"; recBox.style.borderColor = col+"44";

  document.getElementById("recLabel").style.color = col;

  document.getElementById("recLabel").textContent = "TODAY'S SESSION";

  document.getElementById("recText").textContent = sessionRec(composite,fingerR);


  // Systems

  const sysContainer = document.getElementById("systemRows");

  sysContainer.innerHTML = systems.map(sys => {

    const c = statusColor(sys.pct);

    const limiting = sys.pct === minPct;

    return `<div class="sys-row" style="background:var(--card);border-color:${limiting?c+"88":"var(--border)"}">

      <span style="font-size:18px;min-width:28px">${sys.icon}</span>

      <div style="flex:1">

        <div style="font-size:13px;font-weight:600;color:var(--white);margin-bottom:4px">

      <span class="sys-pct" style="color:${c}">${Math.round(sys.pct)}%</span>

  }).join("");


  // Balance

  const balEl = document.getElementById("balanceNum");

  balEl.textContent = (balance>=0?"+":"")+Math.round(balance);

  balEl.style.color = balance>=0?"#22C55E":"#F97316";
